public class Auto {
    String marca;
    String modelo;
    int año;
    double precio;
    boolean disponible;

    public Auto(String marca, String modelo, int año, double precio) {
        this.marca = marca;
        this.modelo = modelo;
        this.año = año;
        this.precio = precio;
        this.disponible = true;
    }
}
